6   package com.hp.autonomy.hod.client.error;
7   
8   import com.hp.autonomy.hod.client.job.Action;
9   import lombok.extern.slf4j.Slf4j;
10  import retrofit.RetrofitError;
11  import retrofit.client.Response;
12  
13  import java.util.List;
14  import java.util.Map;
15  
16  @Slf4j
17  public class DefaultHodErrorHandler implements HodErrorHandler {
18      @Override
19      public Throwable handleError(final RuntimeException cause) {
20          if (!(cause instanceof RetrofitError)) {
21              //noinspection ProhibitedExceptionThrown
22              throw cause;
23          }
24  
25          final RetrofitError retrofitError = (RetrofitError) cause;
26  
27          final RetrofitError.Kind kind = retrofitError.getKind();
28  
29          if (kind == RetrofitError.Kind.HTTP) {
30              // HOD returned an unsuccessful status code, parse the JSON response and throw something better
31              HodError hodError = (HodError) retrofitError.getBodyAs(HodError.class);
32  
33              // HOD errors are tediously non standard
34              // if it has message (rather than reason, it's probably an API key error)
35              final String message = hodError.getMessage();
36  
37              // if has actions, pull the error out of the action as it will be more useful
38              final List<Action<?>> actions = hodError.getActions();
39  
40              int error = hodError.getError();
41  
42              if (message != null && hodError.getErrorCode() == null) {
43                  // HOD has given us an API key error
44                  final Map<String, Object> detail = (Map<String, Object>) hodError.getDetail();
45                  error = (Integer) detail.get("error");
46                  final HodErrorCode errorCode = HodErrorCode.fromCode(error);
47  
48                  hodError = new HodError.Builder()
49                          .setErrorCode(errorCode)
50                          .setReason(message)
51                          .build();
52              }
53              else if (actions != null && !actions.isEmpty()) {
54                  // HOD has given us job style output
55                  hodError = actions.get(0).getErrors().get(0);
56              }
57  
58              // may be null if no response body
59              if (hodError != null) {
60                  if (hodError.getErrorCode() == HodErrorCode.UNKNOWN_ERROR_CODE) {
61                      // a code we've not seen before has been returned, log it
62                      log.error("UNKNOWN ERROR CODE {}", error);
63                  }
64  
65                  final Response response = retrofitError.getResponse();
66                  final int status = response.getStatus();
67  
68                  log.error("{} error communicating with HP Haven OnDemand", status);
69  
70                  log.error("Detail was : {}", hodError.getDetail());
71  
72                  return new HodErrorException(hodError, status);
73              }
74              else {
75                  // no response body, not much we can do
76                  return cause;
77              }
78          }
79          else {
80              // something unusual went wrong, just rethrow the cause
81              return cause;
82          }
83      }
84  }
