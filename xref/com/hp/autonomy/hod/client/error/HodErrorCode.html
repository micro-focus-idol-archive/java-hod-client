195     private static final Map<Integer, HodErrorCode> LOOKUP = new HashMap<>();
196 
197     static {
198         for (final HodErrorCode hodErrorCode : values()) {
199             LOOKUP.put(hodErrorCode.code, hodErrorCode);
200         }
201     }
202 
203     private final int code;
204 
205     HodErrorCode(final int code) {
206         this.code = code;
207     }
208 
209     public static HodErrorCode fromCode(final int code) {
210         final HodErrorCode errorCode = LOOKUP.get(code);
211 
212         if (errorCode != null) {
213             return errorCode;
214         }
215         else {
216             return UNKNOWN_ERROR_CODE;
217         }
218     }
