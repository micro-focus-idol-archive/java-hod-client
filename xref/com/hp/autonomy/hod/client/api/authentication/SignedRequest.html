22  /**
23   * Represents a request to HP Haven OnDemand which must be made from a user's browser. The token header should be set to
24   * the value of the {@link #token} field. If there is a body, the Content-Type header should be set to "application/x-www-form-urlencoded".
25   */
26  @Data
27  public class SignedRequest {
28      private static final String UTF8 = "UTF-8";
29  
30      private final String url;
31      private final Request.Verb verb;
32      private final String body;
33      private final String token;
34  
35      private SignedRequest(final String url, final Request.Verb verb, final String body, final String token) {
36          this.url = url;
37          this.verb = verb;
38          this.body = body;
39          this.token = token;
40      }
41  
42      static SignedRequest sign(final Hmac hmac, final String endpoint, final AuthenticationToken<?, TokenType.HmacSha1> token, final Request<String, String> request) {
43          final URIBuilder uriBuilder;
44  
45          try {
46              uriBuilder = new URIBuilder(endpoint + request.getPath());
47          } catch (final URISyntaxException e) {
48              throw new IllegalArgumentException("Invalid endpoint or request path");
49          }
50  
51          if (request.getQueryParameters() != null) {
52              for (final Map.Entry<String, List<String>> entry : request.getQueryParameters().entrySet()) {
53                  for (final String value : entry.getValue()) {
54                      uriBuilder.addParameter(entry.getKey(), value);
55                  }
56              }
57          }
58  
59          final String bodyString;
60  
61          if (request.getBody() == null) {
62              bodyString = null;
63          } else {
64              final List<NameValuePair> pairs = new LinkedList<>();
65  
66              for (final Map.Entry<String, List<String>> entry : request.getBody().entrySet()) {
67                  pairs.addAll(entry.getValue().stream().map(value -> new BasicNameValuePair(entry.getKey(), value)).collect(Collectors.toList()));
68              }
69  
70              bodyString = URLEncodedUtils.format(pairs, UTF8);
71          }
72  
73          final String tokenString = hmac.generateToken(request, token);
74          return new SignedRequest(uriBuilder.toString(), request.getVerb(), bodyString, tokenString);
75      }
