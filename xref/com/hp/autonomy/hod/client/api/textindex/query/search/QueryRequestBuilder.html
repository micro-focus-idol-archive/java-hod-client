169     /**
170      * @return A map of query parameters suitable for use with {@link QueryTextIndexBackend}. get is NOT supported on
171      * the resulting map
172      */
173     Map<String, Object> build() {
174         final Map<String, Object> map = new MultiMap<>();
175         map.put("end_tag", endTag);
176         map.put("field_text", fieldText);
177         map.put("highlight", highlight);
178         map.put("absolute_max_results", absoluteMaxResults);
179         map.put("max_page_results", maxPageResults);
180         map.put("min_score", minScore);
181         map.put("print", print);
182         map.put("print_fields", StringUtils.join(printFields, ','));
183         map.put("sort", sort);
184         map.put("start", start);
185         map.put("start_tag", startTag);
186         map.put("summary", summary);
187         map.put("total_results", totalResults);
188         map.put("query_profile", queryProfile);
189         map.put("promotion", promotions);
190         map.put("check_spelling", checkSpelling);
191         map.put("security_info", securityInfo);
192 
193         // prefer the DateTime over the numeric versions
194         map.putAll(TimeSelector.max(maxDate, maxDateDays, maxDateSeconds));
195         map.putAll(TimeSelector.min(minDate, minDateDays, minDateSeconds));
196 
197         for (final ResourceIdentifier index : indexes) {
198             map.put("indexes", index);
199         }
200 
201         return map;
202     }
