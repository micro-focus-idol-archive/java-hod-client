9   import com.hp.autonomy.hod.client.api.textindex.query.search.Summary;
10  import com.hp.autonomy.hod.client.util.MultiMap;
11  import lombok.Data;
12  import lombok.experimental.Accessors;
13  import org.apache.commons.lang.StringUtils;

15  import java.util.List;
16  import java.util.Map;

18  /**
19   * Helper class for building up optional parameters for the Get Content API. The default value for all parameters
20   * is null. Null parameters will not be sent to HP Haven OnDemand
21   */
22  @Data
23  @Accessors(chain = true)
24  public class GetContentRequestBuilder {
25  
26      /**
27       * @param highlightExpression Value for the highlight_expression parameter
28       */
29      private String highlightExpression;
30  
31      /**
32       * @param startTag Value for the start_tag parameter
33       */
34      private String startTag;
35  
36      /**
37       * @param endTag Value for the end_tag parameter
38       */
39      private String endTag;
40  
41      /**
42       * @param printFields Value for the print_fields parameter. This list will be joined with commas before being sent
43       * to the server
44       */
45      private List<String> printFields;
46  
47      /**
48       * @param print Value for the print parameter
49       */
50      private Print print;
51  
52      /**
53       * @param summary Value for the summary parameter
54       */
55      private Summary summary;
56  
57      private String securityInfo;
58  
59      /**
60       * @return A map of query parameters suitable for use with {@link GetContentBackend}. get is NOT supported on
61       * the resulting map
62       */
63      Map<String, Object> build() {
64          final Map<String, Object> map = new MultiMap<>();
65          map.put("highlight_expression", highlightExpression);
66          map.put("start_tag", startTag);
67          map.put("end_tag", endTag);
68          map.put("print", print);
69          map.put("print_fields", StringUtils.join(printFields, ','));
70          map.put("summary", summary);
71          map.put("security_info", securityInfo);
72  
73          return map;
74      }
75  
76  }
