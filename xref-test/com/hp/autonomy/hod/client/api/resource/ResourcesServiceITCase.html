32  @RunWith(Parameterized.class)
33  public class ResourcesServiceITCase extends AbstractHodClientIntegrationTest {
34      private ResourcesService resourcesService;
35  
36      public ResourcesServiceITCase(final Endpoint endpoint) {
37          super(endpoint);
38      }
39  
40      @Override
41      @Before
42      public void setUp() {
43          super.setUp();
44          resourcesService = new ResourcesServiceImpl(getConfig());
45      }
46  
47      @Test
48      public void listsResources() throws HodErrorException {
49          final ListResourcesRequestBuilder parameters = new ListResourcesRequestBuilder()
50                  .setTypes(ResourceType.of(ResourceType.TEXT_INDEX));
51  
52          final List<ResourceDetails> resources = resourcesService.list(getTokenProxy(), parameters);
53  
54          final List<ResourceDetails> publicIndexes = resources.stream()
55                  .filter(resource -> ResourceName.PUBLIC_INDEXES_DOMAIN.equals(resource.getResource().getDomain()))
56                  .collect(Collectors.toList());
57  
58          assertThat(publicIndexes, is(not(empty())));
59  
60          for (final ResourceDetails publicResource : publicIndexes) {
61              assertThat(publicResource.getType(), is(ResourceType.TEXT_INDEX));
62              assertThat(publicResource.getResource(), is(not(nullValue())));
63          }
64  
65          assertThat(resources, hasResourceWithIdentifier(ResourceName.WIKI_ENG));
66          assertThat(resources, hasResourceWithIdentifier(getPrivateIndex()));
67      }
68  
69      @Test
70      public void listsAllResources() throws HodErrorException {
71          resourcesService.list(getTokenProxy(), new ListResourcesRequestBuilder());
72      }
73  
74      @Test
75      public void listsDomainResource() throws HodErrorException {
76          final String privateDomain = getPrivateIndex().getDomain();
77          final List<ResourceDetails> resources = resourcesService.list(getTokenProxy(), new ListResourcesRequestBuilder().setDomains(Collections.singleton(privateDomain)));
78  
79          final Optional<ResourceDetails> maybeResource = resources.stream()
80                  .filter(resource -> !resource.getResource().getDomain().equals(privateDomain))
81                  .findFirst();
82  
83          assertThat("A resource with the wrong domain was returned", maybeResource, isEmpty());
84      }
85  
86      static class ResourceMatcher extends BaseMatcher<List<ResourceDetails>> {
87          private final ResourceName identifier;
88  
89          private ResourceMatcher(final ResourceName identifier) {
90              this.identifier = identifier;
91          }
92  
93          static ResourceMatcher hasResourceWithIdentifier(final ResourceName identifier) {
94              return new ResourceMatcher(identifier);
95          }
96  
97          @Override
98          public boolean matches(final Object item) {
99              if (!(item instanceof List)) {
100                 return false;
101             }
102 
103             @SuppressWarnings("unchecked")
104             final Optional<ResourceDetails> maybeResource = ((Collection<ResourceDetails>) item).stream()
105                     .filter(resource -> identifier.equals(resource.getResource().getResourceName()))
106                     .findFirst();
107 
108             return maybeResource.isPresent();
109         }
110 
111         @Override
112         public void describeTo(final Description description) {
113             description.appendText("A list containing a resource with identifier: ").appendValue(identifier);
114         }
115     }
