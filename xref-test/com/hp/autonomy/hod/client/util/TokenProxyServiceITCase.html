36  @RunWith(Parameterized.class)
37  public class TokenProxyServiceITCase extends AbstractHodClientIntegrationTest {
38  
39      private QueryTextIndexService<Document> queryTextIndexService;
40  
41      @Override
42      @Before
43      public void setUp() {
44          super.setUp();
45  
46          // Creating the HodServiceConfig requires the TokenProxyService, but the TokenProxyService requires the
47          // TokenRepository, which requires the HodServiceConfig
48          final AtomicReference<TokenProxy<EntityType.Application, TokenType.Simple>> tokenProxyAtomicReference = new AtomicReference<>();
49  
50          final TokenProxyService<EntityType.Application, TokenType.Simple> tokenProxyService = tokenProxyAtomicReference::get;
51  
52          final HodServiceConfig<EntityType.Application, TokenType.Simple> hodServiceConfig = HodServiceConfigFactory.getHodServiceConfig(tokenProxyService, getEndpoint());
53  
54          try {
55              tokenProxyAtomicReference.set(hodServiceConfig.getTokenRepository().insert(getToken()));
56          } catch (final IOException e) {
57              throw new TokenRepositoryException(e);
58          }
59  
60          queryTextIndexService = QueryTextIndexServiceImpl.documentsService(hodServiceConfig);
61      }
62  
63      public TokenProxyServiceITCase(final Endpoint endpoint) {
64          super(endpoint);
65      }
66  
67      @Test
68      public void testInterceptor() throws HodErrorException {
69          final QueryRequestBuilder params = new QueryRequestBuilder()
70              .addIndexes(ResourceName.WIKI_ENG)
71              .setTotalResults(true);
72  
73          final QueryResults<Document> documents = queryTextIndexService.queryTextIndexWithText("*", params);
74  
75          assertThat(documents.getTotalResults(), is(greaterThan(0)));
76      }
